<script>
    // --- DỮ LIỆU APP ---
    let state = {
        wallet: 15000,
        peaceFund: 1500,
        luckyBag: 88046,
        transactions: [
            { name: "Shopee Checkout", amount: 450, icon: "🛒", time: "Today, 14:30" },
            { name: "Taipei Metro", amount: 25, icon: "🚇", time: "Today, 08:15" },
            { name: "Lunch Combo", amount: 150, icon: "🥗", time: "Yesterday" }
        ]
    };

    let chartInstance = null;
    let toastTimeout = null; // Biến xử lý thời gian tắt Toast

    // --- CẬP NHẬT GIAO DIỆN & RENDER LIST ---
    function updateUI() {
        document.getElementById('wallet-balance').innerText = state.wallet.toLocaleString();
        document.getElementById('peace-balance').innerText = state.peaceFund.toLocaleString();
        document.getElementById('lucky-balance').innerText = state.luckyBag.toLocaleString();

        const txHome = document.getElementById('transaction-list-home');
        const txFull = document.getElementById('transaction-list-full');
        txHome.innerHTML = ''; txFull.innerHTML = '';
        
        let txs = state.transactions.slice().reverse();
        
        // Render List
        txs.forEach((tx, index) => {
            const html = `
                <div class="flex items-center justify-between p-3 border-b border-gray-50">
                    <div class="flex items-center gap-3">
                        <div class="w-10 h-10 bg-gray-100 rounded-full flex items-center justify-center text-lg">${tx.icon}</div>
                        <div><p class="text-sm font-bold text-gray-800">${tx.name}</p><p class="text-[10px] text-gray-400">${tx.time}</p></div>
                    </div>
                    <span class="text-sm font-bold text-gray-800">- NT$ ${tx.amount.toLocaleString()}</span>
                </div>
            `;
            if(index < 3) txHome.innerHTML += html;
            txFull.innerHTML += html;
        });

        updateChart();
    }

    // --- VẼ BIỂU ĐỒ TRÒN ---
    function updateChart() {
        const ctx = document.getElementById('expenseChart').getContext('2d');
        let categories = { "Food": 0, "Shopping": 0, "Transport": 0, "Invest/Transfer": 0, "Other": 0 };
        
        state.transactions.forEach(tx => {
            let n = tx.name.toLowerCase();
            if(n.includes('shopee') || n.includes('buy') || n.includes('bag')) categories["Shopping"] += tx.amount;
            else if(n.includes('metro') || n.includes('transport') || n.includes('taxi')) categories["Transport"] += tx.amount;
            else if(n.includes('transfer') || n.includes('deposit') || n.includes('lucky')) categories["Invest/Transfer"] += tx.amount;
            else if(n.includes('food') || n.includes('lunch') || n.includes('coffee') || tx.icon === '🤖' || tx.icon === '🥗') categories["Food"] += tx.amount;
            else categories["Other"] += tx.amount;
        });

        const labels = Object.keys(categories).filter(k => categories[k] > 0);
        const data = labels.map(k => categories[k]);
        const colors = ['#06C755', '#FF9F40', '#36A2EB', '#9966FF', '#C9CBCF'];

        if (chartInstance) {
            chartInstance.data.labels = labels;
            chartInstance.data.datasets[0].data = data;
            chartInstance.update();
        } else {
            chartInstance = new Chart(ctx, {
                type: 'doughnut',
                data: {
                    labels: labels,
                    datasets: [{ data: data, backgroundColor: colors, borderWidth: 0 }]
                },
                options: {
                    responsive: true, maintainAspectRatio: false, cutout: '65%',
                    plugins: { legend: { position: 'bottom', labels: { boxWidth: 12, font: {size: 10} } } }
                }
            });
        }
    }

    // --- ĐIỀU HƯỚNG MÀN HÌNH ---
    function switchScreen(screen) {
        ['home', 'chat', 'lucky', 'history'].forEach(s => {
            document.getElementById(`screen-${s}`).classList.add('hidden');
        });
        document.getElementById(`screen-${screen}`).classList.remove('hidden');
        document.getElementById('bg-header').style.display = screen === 'home' ? 'block' : 'none';
        
        ['home', 'history'].forEach(s => {
            document.getElementById(`nav-${s}`).className = `flex flex-col items-center gap-1 pb-4 ${screen === s ? 'text-[#06C755]' : 'text-gray-400'}`;
        });

        if(screen === 'history') updateChart();
    }

    // --- HIỂN THỊ TOAST ĐÃ FIX LỖI ---
    function showToast(message) {
        const toast = document.getElementById('system-toast');
        toast.innerText = message;
        toast.classList.remove('hidden');
        
        // Reset thời gian để nếu bấm liên tục, thông báo không bị biến mất ngay
        if (toastTimeout) clearTimeout(toastTimeout);
        toastTimeout = setTimeout(() => toast.classList.add('hidden'), 2500);
    }

    // --- CHỨC NĂNG CƠ BẢN ---
    function openTransferModal() { document.getElementById('transfer-modal').classList.remove('hidden'); }
    function closeTransferModal() { document.getElementById('transfer-modal').classList.add('hidden'); }
    
    function submitTransfer() {
        const id = document.getElementById('transfer-id').value.trim();
        const amount = parseInt(document.getElementById('transfer-amount').value);
        if (!id || isNaN(amount) || amount <= 0) return showToast("Invalid ID or amount!");
        if (amount > state.wallet) return showToast("Insufficient balance!");

        state.wallet -= amount;
        state.transactions.push({ name: `Transfer to ${id}`, amount: amount, icon: "💸", time: "Just now" });
        updateUI(); closeTransferModal();
        showToast(`Sent NT$ ${amount} to ${id}`);
        document.getElementById('transfer-id').value = ''; document.getElementById('transfer-amount').value = '';
    }

    function handleMood(mood, amount) {
        if (state.wallet >= amount && amount > 0) {
            state.wallet -= amount; state.peaceFund += amount; updateUI();
        }
        showToast(amount > 0 ? `+${amount} NTD auto-saved to Peace Fund!` : "Balance is key. Have a good day!");
    }

    function handleDeposit() {
        const amount = parseInt(document.getElementById('deposit-input').value);
        if (isNaN(amount) || amount <= 0) return showToast("Invalid amount.");
        if (amount > state.wallet) return showToast("Insufficient balance.");

        state.wallet -= amount; state.luckyBag += amount;
        state.transactions.push({ name: "Deposit to Lucky Bag", amount: amount, icon: "💰", time: "Just now" });
        updateUI(); document.getElementById('deposit-input').value = '';
        showToast(`Deposited NT$ ${amount} to Lucky Bag!`);
    }

    function processChat() {
        const input = document.getElementById('chat-input');
        const history = document.getElementById('chat-history');
        const text = input.value.trim();
        if(!text) return;

        history.innerHTML += `<div class="flex justify-end fade-in"><div class="bg-[#06C755] text-white p-3 rounded-2xl rounded-tr-none max-w-[80%]">${text}</div></div>`;
        input.value = ''; history.scrollTop = history.scrollHeight;

        const matchAmount = text.match(/\d+/);
        const amount = matchAmount ? parseInt(matchAmount[0]) : 0;

        setTimeout(() => {
            let botReply = '';
            if (amount > 10000) {
                botReply = `<span class="text-red-600 font-bold">⚠️ Blocked (99.99% Security)</span><br>NT$ ${amount.toLocaleString()} is unusually high.`;
            } else if (amount > 0 && amount <= state.wallet) {
                state.wallet -= amount;
                state.transactions.push({ name: `AI Logged: ${text.replace(/\d+/g, '').trim() || 'Expense'}`, amount: amount, icon: "🤖", time: "Just now" });
                updateUI();
                botReply = `Logged <b>${amount} NTD</b>. Your wallet has ${state.wallet.toLocaleString()} NTD left.`;
            } else if (amount > state.wallet) {
                botReply = `Failed. Your wallet only has NT$ ${state.wallet.toLocaleString()}.`;
            } else {
                botReply = "I couldn't detect an amount. E.g., 'Coffee 60'.";
            }

            history.innerHTML += `
                <div class="flex items-end gap-2 justify-start fade-in mt-4">
                    <div class="w-8 h-8 bg-[#06C755] text-white rounded-full flex items-center justify-center text-xs shrink-0">🤖</div>
                    <div class="bg-white border-gray-100 text-gray-700 p-3 rounded-2xl rounded-bl-none max-w-[80%] border">${botReply}</div>
                </div>`;
            history.scrollTop = history.scrollHeight;
        }, 600);
    }

    document.getElementById("chat-input").addEventListener("keypress", function(e) {
        if (e.key === "Enter") { e.preventDefault(); processChat(); }
    });

    updateUI();
</script>
